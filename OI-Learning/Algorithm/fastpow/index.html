Matrix matrix_pow(Matrix base, long long exp) {

    if (base.rows != base.cols) {
        // error check
    }

    int n = base.rows;
    // åˆå§‹åŒ–å•ä½çŸ©é˜µ
    Matrix result(n,n);
    for(int i = 0; i < n; i++) result[i][i] = 1;

    // å¿«é€Ÿå¹‚ç®—æ³•
    while (exp > 0) {
        if (exp & 1) {                     // å¦‚æœæŒ‡æ•°æ˜¯å¥‡æ•°
            result = result * base;

        base = base * base;  // åº•æ•°å¹³æ–¹
        exp >>= 1;                         // æŒ‡æ•°é™¤ä»¥2
    }

    return result;
}
